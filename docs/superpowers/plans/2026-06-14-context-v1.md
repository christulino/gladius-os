// runtime/contextEntries.js
import { pool } from '../db/postgres.js'
import { emitEvent } from '../core/events.js'

export async function listContextEntries(workItemId, { types } = {}) {
  const params = [workItemId]
  let typeFilter = ''
  if (types && types.length > 0) {
    params.push(types)
    typeFilter = `AND ce.type = ANY($${params.length})`
  }
  const r = await pool.query(`
    SELECT ce.*, u.display_name AS author_name
    FROM runtime.context_entries ce
    LEFT JOIN blueprint.users u ON u.id = ce.author_id
    WHERE ce.work_item_id = $1 ${typeFilter}
    ORDER BY ce.created_at DESC, ce.id DESC
  `, params)
  return r.rows
}

export async function createContextEntry(workItemId, {
  type, title, content, visibility = 'item', tags = [], authorId, isAgent = false
}, client = null) {
  const db = client || pool
  const needsClient = !client
  const c = needsClient ? await pool.connect() : db

  try {
    if (needsClient) await c.query('BEGIN')

    const r = await c.query(`
      INSERT INTO runtime.context_entries
        (work_item_id, type, title, content, visibility, tags, author_id, is_agent)
      VALUES ($1,$2,$3,$4,$5,$6,$7,$8)
      RETURNING *
    `, [workItemId, type, title || null, content, visibility, tags, authorId || null, isAgent])
    const entry = r.rows[0]

    await emitEvent(c, {
      eventType: 'work_item.context_entry_added',
      entityType: 'work_item',
      entityId: String(workItemId),
      actorId: authorId || null,
      payload: { entry_id: entry.id, type: entry.type, is_agent: isAgent },
    })

    if (needsClient) await c.query('COMMIT')
    return entry
  } catch (err) {
    if (needsClient) await c.query('ROLLBACK')
    throw err
  } finally {
    if (needsClient) c.release()
  }
}

export async function updateContextEntry(entryId, { content, title, tags, visibility }) {
  const sets = [], params = []
  if (content   !== undefined) { params.push(content);    sets.push(`content=$${params.length}`) }
  if (title     !== undefined) { params.push(title);      sets.push(`title=$${params.length}`) }
  if (tags      !== undefined) { params.push(tags);       sets.push(`tags=$${params.length}`) }
  if (visibility!== undefined) { params.push(visibility); sets.push(`visibility=$${params.length}`) }
  if (sets.length === 0) throw new Error('Nothing to update')
  sets.push('is_edited=true', `updated_at=now()`)
  params.push(entryId)
  const r = await pool.query(
    `UPDATE runtime.context_entries SET ${sets.join(',')} WHERE id=$${params.length} RETURNING *`,
    params
  )
  return r.rows[0] || null
}

export async function deleteContextEntry(entryId) {
  const r = await pool.query(
    `DELETE FROM runtime.context_entries WHERE id=$1 RETURNING id`,
    [entryId]
  )
  return r.rowCount > 0
}
```

// Parse YAML frontmatter from playbook content.
// Returns { meta, body } where meta is the parsed frontmatter object.
export function parsePlaybook(content) {
  const match = content.match(/^---\n([\s\S]*?)\n---\n?([\s\S]*)$/)
  if (!match) return { meta: {}, body: content }
  try {
    // js-yaml v4: yaml.load() is the safe API (safeLoad was deprecated).
    // Playbook frontmatter is plain data — no custom tags, no arbitrary execution.
    const meta = yaml.load(match[1], { schema: yaml.JSON_SCHEMA }) || {}
    return { meta, body: match[2].trim() }
  } catch {
    return { meta: {}, body: content }
  }
}

// runtime/orgAiModels.js
import crypto from 'node:crypto'
import { pool } from '../db/postgres.js'

const ALGO = 'aes-256-gcm'
const KEY_HEX = process.env.FLOWOS_ENCRYPTION_KEY
// Key must be 32 bytes hex (64 hex chars). Generate with: openssl rand -hex 32

function getKey() {
  if (!KEY_HEX) throw new Error('FLOWOS_ENCRYPTION_KEY env var not set')
  return Buffer.from(KEY_HEX, 'hex')
}

export function encryptApiKey(plaintext) {
  const key = getKey()
  const iv = crypto.randomBytes(12)
  const cipher = crypto.createCipheriv(ALGO, key, iv)
  const enc = Buffer.concat([cipher.update(plaintext, 'utf8'), cipher.final()])
  const tag = cipher.getAuthTag()
  return [iv.toString('hex'), enc.toString('hex'), tag.toString('hex')].join('.')
}

export function decryptApiKey(ciphertext) {
  const key = getKey()
  const [ivHex, encHex, tagHex] = ciphertext.split('.')
  const decipher = crypto.createDecipheriv(ALGO, key, Buffer.from(ivHex, 'hex'))
  decipher.setAuthTag(Buffer.from(tagHex, 'hex'))
  const dec = Buffer.concat([decipher.update(Buffer.from(encHex, 'hex')), decipher.final()])
  return dec.toString('utf8')
}

export async function listOrgAiModels(orgId) {
  const r = await pool.query(
    `SELECT id,org_id,name,provider,model,is_active,created_at,updated_at
     FROM blueprint.org_ai_models WHERE org_id=$1 ORDER BY name`,
    [orgId]
  )
  return r.rows  // api_key_enc never returned to client
}

export async function createOrgAiModel(orgId, { name, provider, model, apiKey }) {

    VALUES ($1,$2,$3,$4,$5)

export async function updateOrgAiModel(id, { name, provider, model, apiKey, isActive }) {

// Minimal markdown renderer: handles headers, bold, code, lists, tables, line breaks
function MarkdownBody({ content }) {
  if (!content) return null
  const lines = content.split('\n')
  const elements = []
  let i = 0

  while (i < lines.length) {
    const line = lines[i]

    // Table
    if (line.startsWith('|') && lines[i+1]?.match(/^\|[-| :]+\|$/)) {
      const headers = line.split('|').filter(Boolean).map(h => h.trim())
      i += 2
      const rows = []
      while (i < lines.length && lines[i].startsWith('|')) {
        rows.push(lines[i].split('|').filter(Boolean).map(c => c.trim()))
        i++
      }
      elements.push(
        <table key={i} className="w-full text-xs border-collapse my-2">
          <thead>
            <tr>{headers.map((h,j) => <th key={j} className="border border-border px-2 py-1 text-left bg-muted font-semibold">{h}</th>)}</tr>
          </thead>
          <tbody>
            {rows.map((row,ri) => (
              <tr key={ri} className={ri%2===0?'':'bg-muted/30'}>
                {row.map((cell,ci) => <td key={ci} className="border border-border px-2 py-1">{cell}</td>)}
              </tr>
            ))}
          </tbody>
        </table>
      )
      continue
    }

    // H2/H3
    if (line.startsWith('### ')) {
      elements.push(<h3 key={i} className="text-xs font-bold text-foreground mt-3 mb-1">{line.slice(4)}</h3>)
    } else if (line.startsWith('## ')) {
      elements.push(<h2 key={i} className="text-sm font-bold text-foreground mt-3 mb-1">{line.slice(3)}</h2>)
    } else if (line.startsWith('- ') || line.match(/^\d+\. /)) {
      const text = line.replace(/^[-\d]+\.?\s/, '')
      elements.push(<li key={i} className="text-xs text-foreground ml-3 list-disc">{renderInline(text)}</li>)
    } else if (line.trim() === '') {
      elements.push(<div key={i} className="h-1" />)
    } else {
      elements.push(<p key={i} className="text-xs text-foreground leading-relaxed">{renderInline(line)}</p>)
    }
    i++
  }
  return <div className="flex flex-col gap-0.5">{elements}</div>
}

function renderInline(text) {
  // Bold + code — simple split
  const parts = text.split(/(`[^`]+`|\*\*[^*]+\*\*)/)
  return parts.map((p, i) => {
    if (p.startsWith('`') && p.endsWith('`'))
      return <code key={i} className="bg-muted px-1 rounded text-[10px] font-mono">{p.slice(1,-1)}</code>
    if (p.startsWith('**') && p.endsWith('**'))
      return <strong key={i} className="font-semibold">{p.slice(2,-2)}</strong>
    return p
  })
}

// runtime/contextAssembly.js
import { pool } from '../db/postgres.js'

/**
 * Assemble the full context bundle for a playbook execution.
 * Returns { orgContext, ancestorEntries, itemEntries }
 */
export async function assembleContext(workItemId, orgId, meta) {
  const pullTypes = (meta.context?.pull || []).map(p => p.type)
  const orgTypes  = meta.context?.org || []

  // 1. Org context
  let orgContext = []
  if (orgTypes.length > 0) {
    const r = await pool.query(`
      SELECT type, title, content FROM blueprint.org_context
      WHERE org_id=$1 AND type=ANY($2)
      ORDER BY type, created_at
    `, [orgId, orgTypes])
    orgContext = r.rows
  }

  // 2. Ancestor context (entries with visibility='descendants' from parent items)
  const ancestorPullTypes = (meta.context?.pull || [])
    .filter(p => p.traverse === 'ancestors')
    .map(p => p.type)

  let ancestorEntries = []
  if (ancestorPullTypes.length > 0) {
    const r = await pool.query(`
      WITH RECURSIVE ancestors(id, depth) AS (
        SELECT parent_id, 1
        FROM runtime.work_items WHERE id=$1 AND parent_id IS NOT NULL
        UNION ALL
        SELECT wi.parent_id, a.depth+1
        FROM runtime.work_items wi
        JOIN ancestors a ON wi.id=a.id
        WHERE wi.parent_id IS NOT NULL
      )
      SELECT ce.type, ce.title, ce.content, a.depth
      FROM runtime.context_entries ce
      JOIN ancestors a ON ce.work_item_id=a.id
      WHERE ce.type=ANY($2) AND ce.visibility='descendants'
      ORDER BY a.depth ASC, ce.created_at ASC
    `, [workItemId, ancestorPullTypes])
    ancestorEntries = r.rows
  }

  // 3. Item's own context entries
  let itemEntries = []
  if (pullTypes.length > 0) {
    const r = await pool.query(`
      SELECT type, title, content FROM runtime.context_entries
      WHERE work_item_id=$1 AND type=ANY($2)
      ORDER BY created_at ASC
    `, [workItemId, pullTypes])
    itemEntries = r.rows
  }

  return { orgContext, ancestorEntries, itemEntries }
}

/** Format the assembled bundle as a string for the AI user message */
export function formatBundle(workItem, { orgContext, ancestorEntries, itemEntries }, playbookBody) {
  const parts = []

  if (orgContext.length > 0) {
    parts.push('# Org Context')
    for (const e of orgContext) {
      parts.push(`## [${e.type}] ${e.title}\n${e.content}`)
    }
  }

  if (ancestorEntries.length > 0) {
    parts.push('# Context from Parent Items')
    for (const e of ancestorEntries) {
      parts.push(`## [${e.type}]${e.title ? ' ' + e.title : ''}\n${e.content}`)
    }
  }

  if (itemEntries.length > 0) {
    parts.push('# This Item\'s Context')
    for (const e of itemEntries) {
      parts.push(`## [${e.type}]${e.title ? ' ' + e.title : ''}\n${e.content}`)
    }
  }

  parts.push('# Work Item')
  parts.push(`Title: ${workItem.title}\nType: ${workItem.type_name || 'unknown'}`)

  parts.push('# Your Instructions')
  parts.push(playbookBody)

  return parts.join('\n\n')
}

import { getPlaybookForStage, parsePlaybook } from './stagePlaybooks.js'
import { assembleContext, formatBundle } from './contextAssembly.js'
import { createContextEntry } from './contextEntries.js'

/**
 * Fire the playbook for a stage entry/exit, if one exists.
 * Called post-commit — never throws (logs only).
 */
export async function firePlaybook(workItemId, stageId, witTypeId, orgId, trigger = 'on_enter') {
  try {
    const playbook = await getPlaybookForStage(stageId, witTypeId)
    if (!playbook || !playbook.is_active) return

    const { meta, body } = parsePlaybook(playbook.content)
    if (!meta.trigger || meta.trigger !== trigger) return
    if (!body.trim()) return

    const modelName = meta.model || 'default'
    const cfg = await resolveModelConfig(orgId, modelName)
    if (!cfg || !cfg.apiKey) {
      console.warn(`[playbookExecutor] No model config "${modelName}" for org ${orgId}`)
      return
    }

    // Fetch work item for context
    const wi = await pool.query(
      `SELECT wi.*, wit.name AS type_name FROM runtime.work_items wi
       LEFT JOIN blueprint.work_item_types wit ON wit.id=wi.work_item_type_id
       WHERE wi.id=$1`, [workItemId]
    )
    if (!wi.rows[0]) return
    const workItem = wi.rows[0]

    const bundle  = await assembleContext(workItemId, orgId, meta)
    const userMsg = formatBundle(workItem, bundle, body)

    const writeAllowlist = new Set(meta.context?.write || [])
    if (writeAllowlist.size === 0) {
      console.warn(`[playbookExecutor] Playbook ${playbook.id} has no write allowlist — skipping AI call`)
      return
    }

    const client = new Anthropic({ apiKey: cfg.apiKey })

    const tools = [{
      name: 'write_context_entry',
      description: 'Write a context entry back to the work item journal.',
      input_schema: {
        type: 'object',
        properties: {
          type:       { type: 'string', description: `One of: ${[...writeAllowlist].join(', ')}` },
          title:      { type: 'string', description: 'Short label (optional)' },
          content:    { type: 'string', description: 'Markdown content' },
          visibility: { type: 'string', enum: ['item','descendants'], default: 'item' },
        },
        required: ['type','content'],
      },
    }]

    const resp = await client.messages.create({
      model:      cfg.model,
      max_tokens: 2048,
      system:     'You are a workflow agent. Use write_context_entry to write your output back to the work item. Do not invent facts — only use context provided.',
      messages:   [{ role: 'user', content: userMsg }],
      tools,
    }, { timeout: 60_000 })

    for (const block of resp.content) {
      if (block.type !== 'tool_use' || block.name !== 'write_context_entry') continue
      const { type, title, content, visibility = 'item' } = block.input
      if (!writeAllowlist.has(type)) {
        console.warn(`[playbookExecutor] Blocked write to type "${type}" — not in allowlist`)
        continue
      }
      await createContextEntry(workItemId, { type, title, content, visibility, isAgent: true })
    }
  } catch (err) {
    console.error('[playbookExecutor] Error:', err.message)
  }
}
```

// Run: node mcp/server.js
// Add to Claude Desktop: { "command": "node", "args": ["/path/to/flowos/mcp/server.js"] }

import { Server }            from '@modelcontextprotocol/sdk/server/index.js'
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js'
import { CallToolRequestSchema, ListToolsRequestSchema } from '@modelcontextprotocol/sdk/types.js'
import { pool }              from '../db/postgres.js'
import { listContextEntries, createContextEntry } from '../runtime/contextEntries.js'
import { listOrgContext }    from '../runtime/orgContext.js'

const SYSTEM_PROMPT = `You are connected to FlowOS, an AI-native work management system.

Key concepts:
- Work items move through stages. Transitions are gated by exit criteria.
- Context accumulates as journal entries (type: nfr, discovery, acceptance, design, note, etc.)
- entries with visibility "descendants" are inherited by child work items
- Use get_context() before acting on any item — don't assume what context exists
- Use search() to find related work (JQL syntax: title ~ "auth", status = "In Progress")
- Prefer add_context_entry() over add_comment() when writing something another agent will need to read
- Agents are users. All actions require authentication via API key (fos_ak_ prefix) passed as env var FLOWOS_API_KEY`

const server = new Server(
  { name: 'flowos', version: '1.0.0' },
  { capabilities: { tools: {} } }
)

// ── Tool Definitions ──────────────────────────────────────────────────────────

server.setRequestHandler(ListToolsRequestSchema, async () => ({
  tools: [
    {
      name: 'get_work_item',
      description: 'Get a work item by ID including current stage, fields, and metadata.',
      inputSchema: { type:'object', properties:{ id:{type:'number'} }, required:['id'] },
    },
    {
      name: 'search',
      description: 'Search work items using JQL. Examples: title ~ "auth", status = "In Progress", type = "Bug", assignee = me',
      inputSchema: { type:'object', properties:{ jql:{type:'string'}, limit:{type:'number'} }, required:['jql'] },
    },
    {
      name: 'get_context',
      description: 'Get journal context entries for a work item. Optionally filter by type and traverse ancestors.',
      inputSchema: {
        type:'object',
        properties:{
          work_item_id: {type:'number'},
          types:        {type:'array', items:{type:'string'}, description:'e.g. ["nfr","discovery"]'},
          traverse:     {type:'string', enum:['none','ancestors'], description:'ancestors = include parent item entries with visibility:descendants'},
        },
        required:['work_item_id'],
      },
    },
    {
      name: 'add_context_entry',
      description: 'Write a context entry to a work item journal. Use for structured knowledge (requirements, decisions, notes) that other agents will read.',
      inputSchema: {
        type:'object',
        properties:{
          work_item_id: {type:'number'},
          type:         {type:'string', description:'nfr|discovery|acceptance|design|decision|note|test-plan'},
          title:        {type:'string'},
          content:      {type:'string', description:'Markdown content'},
          visibility:   {type:'string', enum:['item','descendants'], description:'descendants = visible to child work items'},
        },
        required:['work_item_id','type','content'],
      },
    },
    {
      name: 'add_comment',
      description: 'Add a comment to a work item. Use for human-readable discussion, not structured data.',

          types:  {type:'array', items:{type:'string'}},

        required:['org_id'],

    },

      name: 'list_stages',
      description: 'List stages in a workflow.',
      inputSchema: { type:'object', properties:{ workflow_id:{type:'number'} }, required:['workflow_id'] },

      name: 'run_playbook',
      description: 'Manually trigger the playbook for a stage against a work item. The item does not need to currently be in that stage.',
      inputSchema: {
        type:'object',

          work_item_id:{type:'number'},
          stage_id:    {type:'number'},
        },
        required:['work_item_id','stage_id'],
      },

// ── Tool Handlers ─────────────────────────────────────────────────────────────

server.setRequestHandler(CallToolRequestSchema, async (req) => {
  const { name, arguments: args } = req.params

  try {
    switch (name) {

      case 'get_work_item': {
        const r = await pool.query(`
          SELECT wi.*, s.name AS stage_name, wit.name AS type_name, o.name AS org_name
          FROM runtime.work_items wi
          LEFT JOIN blueprint.stages s ON s.id=wi.current_stage_id
          LEFT JOIN blueprint.work_item_types wit ON wit.id=wi.work_item_type_id
          LEFT JOIN blueprint.organizations o ON o.id=wi.owner_org_id
          WHERE wi.id=$1
        `, [args.id])
        if (!r.rows[0]) return { content:[{type:'text',text:`Work item ${args.id} not found`}] }
        return { content:[{type:'text',text:JSON.stringify(r.rows[0],null,2)}] }
      }

      case 'search': {
        // Simple title search for v1 — full JQL engine hookup is follow-up
        const limit = Math.min(args.limit || 20, 50)
        const r = await pool.query(`
          SELECT wi.id, wi.display_key, wi.title, s.name AS stage, wit.name AS type
          FROM runtime.work_items wi
          LEFT JOIN blueprint.stages s ON s.id=wi.current_stage_id
          LEFT JOIN blueprint.work_item_types wit ON wit.id=wi.work_item_type_id
          WHERE wi.title ILIKE $1
          LIMIT $2
        `, [`%${args.jql.replace(/title\s*~\s*"?([^"]+)"?/i,'$1')}%`, limit])
        return { content:[{type:'text',text:JSON.stringify({rows:r.rows,count:r.rowCount})}] }
      }

      case 'get_context': {
        const rows = await listContextEntries(args.work_item_id, { types: args.types })
        // TODO: add ancestor traversal using assembleContext when args.traverse === 'ancestors'
        return { content:[{type:'text',text:JSON.stringify({rows,count:rows.length})}] }
      }

      case 'add_context_entry': {
        const entry = await createContextEntry(args.work_item_id, {
          type: args.type,
          title: args.title,
          content: args.content,
          visibility: args.visibility || 'item',
          isAgent: true,
        })
        return { content:[{type:'text',text:JSON.stringify(entry)}] }
      }

      case 'add_comment': {
        const r = await pool.query(`
          INSERT INTO runtime.work_item_comments (work_item_id, body, author_user_id)
          VALUES ($1,$2,NULL) RETURNING id, body, created_at
        `, [args.work_item_id, args.body])
        return { content:[{type:'text',text:JSON.stringify(r.rows[0])}] }
      }

      case 'get_org_context': {
        const rows = await listOrgContext(args.org_id, { types: args.types })
        return { content:[{type:'text',text:JSON.stringify({rows,count:rows.length})}] }
      }

      case 'list_stages': {
        const r = await pool.query(
          `SELECT id, name, stage_class, display_order FROM blueprint.stages WHERE workflow_id=$1 ORDER BY display_order`,
          [args.workflow_id]
        )
        return { content:[{type:'text',text:JSON.stringify({rows:r.rows})}] }
      }

      case 'run_playbook': {
        const wi = await pool.query(
          `SELECT owner_org_id, work_item_type_id FROM runtime.work_items WHERE id=$1`,
          [args.work_item_id]
        )
        if (!wi.rows[0]) return { content:[{type:'text',text:'Work item not found'}] }
        const { owner_org_id, work_item_type_id } = wi.rows[0]
        await firePlaybook(args.work_item_id, args.stage_id, work_item_type_id, owner_org_id, 'manual')
        return { content:[{type:'text',text:`Playbook triggered for work item ${args.work_item_id} on stage ${args.stage_id}`}] }
      }

      default:
        return { content:[{type:'text',text:`Unknown tool: ${name}`}], isError:true }
    }
  } catch (err) {
    return { content:[{type:'text',text:`Error: ${err.message}`}], isError:true }
  }
})

// ── Start ─────────────────────────────────────────────────────────────────────

const transport = new StdioServerTransport()
await server.connect(transport)
console.error(`FlowOS MCP server running (stdio)\n${SYSTEM_PROMPT}`)
